% ─── METHOD A ────────────────────────────────────────────────────────────────

\begin{frame}{Method A --- Co-occurrence}
  \begin{twopanelrow}
    \twopanel{How it works}{
      \small
      \begin{enumerate}
        \item For each source animal, iterate over sentences in its Wikipedia page.
        \item If any target animal is mentioned in the \alert{same sentence}: add edge source $\to$ target.
        \item No verb, no syntactic analysis --- pure proximity.
      \end{enumerate}
      \vspace{0.4em}
      \textbf{Example sentence:}\\
      \textit{``Foxes and rabbits both live in the forest.''}\\[0.3em]
      $\Rightarrow$ adds Fox$\to$Rabbit \\
      \phantom{$\Rightarrow$ }even though the sentence is about their \emph{habitat}.
    }
    \twopanel{Design trade-offs}{
      \small
      \good: highest possible recall --- every real edge that co-occurs
      in a sentence will be found.

      \vspace{0.5em}
      \bad: no predation signal is required. Any two animals appearing
      together (as competitors, taxonomic relatives, or examples) become
      an edge.

      \vspace{0.5em}
      \bad: no direction inference --- edges are always assigned as
      source $\to$ target based on which page is being read.

      \vspace{0.5em}
      \textbf{Expected:} very high recall, low precision.
    }
  \end{twopanelrow}
  \note{
    Present Method A as the recall ceiling. Its result tells us how many gold
    edges are even textually accessible from these Wikipedia pages.
  }
\end{frame}

% ─── METHOD B ────────────────────────────────────────────────────────────────

\begin{frame}{Method B --- Predation Verb Patterns}
  \begin{twopanelrow}
    \twopanel{How it works}{
      \small
      \begin{enumerate}
        \item For each sentence containing the source animal, extract a window around both animals.
        \item Requires at least one \alert{predation verb or phrase} inside that window:\\
            
        \item \alert{Infer direction:}
          \begin{itemize}
            \item Active: ``Fox \emph{hunts} Rabbit'' $\to$ Fox$\to$Rabbit.
            \item Passive: ``Rabbit \emph{is hunted by} Fox'' \\  Fox$\to$Rabbit.
          \end{itemize}
          \item \alert{Category expansion fallback}: ``eats insects'', expands to ``eats fly``
      \end{enumerate}
    }
    \twopanel{Design trade-offs}{
      \small
      \good: the verb requirement dramatically reduces false positives
      compared to Method~A.

      \vspace{0.5em}
      \bad: paraphrased predation (\emph{``diet consists of\ldots''},
      \emph{``food source for\ldots''}) is missed, lowering recall.

      \vspace{0.5em}
      \bad: direction inference from text position is heuristic; complex
      sentence structures may be mis-directed.

      \vspace{0.5em}
      \bad: plural forms (mice, flies) not handled uniformly in the
      window search.

      \vspace{0.5em}
      \textbf{Expected:} highest precision, moderate recall.
    }
  \end{twopanelrow}
  \note{
    Method B is the precision anchor. It shows that insisting on an explicit
    verb is a necessary but not sufficient condition for high F1.
  }
\end{frame}

% ─── METHOD C ────────────────────────────────────────────────────────────────

\begin{frame}{Method C --- OpenIE-style Relation Extraction}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.56\textwidth}
      \small
      \textbf{Core idea:} extract \texttt{(subject, relation, object)} triples.

      \vspace{0.4em}
      \begin{block}{Step-by-step}
        \begin{enumerate}
          \item Build regex from the full predation phrase + keyword vocabulary.
          \item Search for \texttt{[source] \ldots [rel] \ldots [target]} up to 180 characters apart.
          \item Handle passive: \texttt{[target] \ldots is eaten by \ldots [source]}.
          \item \alert{Category expansion fallback}: ``eats insects'', expands to ``eats fly``
        \end{enumerate}
      \end{block}
    \column{0.40\textwidth}
      \small
      \good: variable-distance regex captures paraphrases missed by Method~B's fixed window.

      \vspace{0.5em}
      \good: category expansion recovers edges when Wikipedia mentions a category (``insects'') rather than a specific animal.

      \vspace{0.5em}
      \bad: generous distance windows may allow subject--relation--object triples that are not actually related.

      \vspace{0.5em}
      \bad: unconstrained category expansion can introduce false positives for all members of a category.

      \vspace{0.5em}
      \textbf{Expected:} better recall than B, moderate precision.
  \end{columns}
  \note{
    Emphasise the category expansion idea --- it is the main novel step over
    Method B, and the main source of both extra recall and extra false positives.
  }
\end{frame}

\begin{frame}{Method D --- Hybrid Extraction (Overview)}

\small

Method D combines rule-based extraction with constrained fallback expansion:

\begin{itemize}
    \item \textbf{Step 1: Explicit relation extraction}
    \begin{itemize}
        \item Predicate-driven extraction using verb cues + proximity constraints
    \end{itemize}

    \item \textbf{Step 2: Restricted category expansion}
    \begin{itemize}
        \item Applied only if Step 1 fails
        \item Uses insect-based semantic expansion under strict constraints
    \end{itemize}
\end{itemize}

\vspace{0.6em}

\textbf{Key idea:} maximize precision first, then recover recall via controlled fallback.
\end{frame}

\begin{frame}{Method D --- Step 1: Explicit Relation Extraction}

\small

\textbf{Goal:} extract high-precision predator $\rightarrow$ prey edges

\vspace{0.4em}

\textbf{Procedure:}

\begin{itemize}
    \item Identify predation verbs (eat, hunt, kill, consume, attack, etc.)
    \item Reject sentences containing negation cues (e.g. not, rarely, never)
    \item Detect entity mentions with lexical variants (e.g. mouse/mice)
\end{itemize}

\vspace{0.4em}

\textbf{Edge creation conditions:}
\begin{itemize}
    \item Both predator and prey appear in the same sentence
    \item Both are within a token distance threshold to predation verbs:
    \begin{itemize}
        \item predator distance $\leq d_{pred}$
        \item prey distance $\leq d_{prey}$
    \end{itemize}
    \item OR passive construction detected (``X is eaten by Y'')
\end{itemize}

\vspace{0.4em}

\textbf{Effect:} high precision extraction with strict linguistic constraints

\end{frame}

\begin{frame}{Method D --- Step 2: Restricted Category Expansion}

\small

\textbf{Goal:} recover missed edges from Step 1 under controlled conditions

\vspace{0.4em}

\textbf{Trigger condition:}
\begin{itemize}
    \item Only executed if Step 1 does not extract any relation
    \item Sentence must contain at least one predation cue
\end{itemize}

\vspace{0.4em}

\textbf{Expansion strategy:}
\begin{itemize}
    \item Expand prey candidates using insect taxonomy only
    \item Apply only to insectivorous source species:
    \begin{itemize}
        \item Salamander, Lizard, Spider, Bat, Bird
    \end{itemize}
\end{itemize}

\vspace{0.4em}

\textbf{Effect:}
\begin{itemize}
    \item Improves recall without significantly increasing false positives
    \item Acts as a safety-constrained fallback mechanism
\end{itemize}

\end{frame}





\begin{frame}{Method D --- Strengths \& Limitations}

\begin{columns}[T,totalwidth=\textwidth]

\column{0.5\textwidth}

\textbf{Strengths}
\begin{itemize}
    \item Combines explicit linguistic rules with controlled fallback expansion
    \item Dual proximity constraint (predator + prey) improves precision
    \item Handles passive constructions (e.g. ``X is eaten by Y'')
    \item Lexical variant matching improves robustness (mouse/mice, etc.)
    \item Negation filtering reduces false positive relations
\end{itemize}

\column{0.5\textwidth}

\textbf{Limitations}
\begin{itemize}
    \item Relies on manually defined predation vocabulary
    \item Token-based distance is only a proxy for syntactic structure
    \item Category expansion is restricted to insect prey
    \item Cannot capture cross-sentence relations
    \item Performance depends on sentence-level parsing quality
\end{itemize}

\end{columns}

\vspace{0.5em}

\centering
\textbf{Overall:} best-performing hybrid trade-off between precision and recall

\end{frame}

% ─── EVALUATION SETUP ────────────────────────────────────────────────────────

\begin{frame}{Evaluation Metrics}
  \begin{twopanelrow}
    \twopanel{Edge-level evaluation}{
      Let $\hat{E}$ be the set of predicted edges and $E^{*}$ the gold standard.
      \[
        \text{Precision} = \frac{|\hat{E} \cap E^{*}|}{|\hat{E}|}
        %\qquad
        \text{ Recall} = \frac{|\hat{E} \cap E^{*}|}{|E^{*}|}
      \]
      \[
        F_1 = \frac{2 \cdot P \cdot R}{P + R}
      \]
      \begin{itemize}
        \item $|\hat{E} \cap E^{*}|$ = true positives (TP)
        \item $|\hat{E}| - \text{TP}$ = false positives (FP)
        \item $|E^{*}| - \text{TP}$ = false negatives (FN)
      \end{itemize}
    }
    \twopanel{Scope constraint}{
      \small
      Only edges that satisfy the trophic direction are considered valid:
      \vspace{0.3em}
      \begin{itemize}
        \item Source must be in \textcolor{td3red}{top predators} $\cup$ \textcolor{td3gold}{mid-level}.
        \item Target must be in \textcolor{td3gold}{mid-level} $\cup$ \textcolor{td3green}{prey}.
        \item Self-loops are never valid.
        \item Top predators cannot appear as prey targets.
      \end{itemize}
      \vspace{0.3em}
      All four methods use this same constraint, implemented via \texttt{sanitize\_edges()}.
    }
  \end{twopanelrow}
  \note{
    Explain the scope constraint clearly. It is important for the audience
    to understand that we never penalise a method for not predicting biologically
    impossible edges.
  }
\end{frame}

% ─── ERROR ANALYSIS ──────────────────────────────────────────────────────────
\begin{frame}{Error Analysis}
\small

\begin{itemize}
  \item \textbf{Method A --- Co-occurrence}
  \begin{itemize}
    \item High recall due to saturation strategy
    \item Very large number of false positives (semantic noise)
    \item Lacks directional and relational constraints
  \end{itemize}

  \item \textbf{Method B --- Dependency Patterns}
  \begin{itemize}
    \item Very low recall due to strict pattern dependency
    \item Fails on lexical variation and non-canonical syntax
    \item High precision but extremely sparse predictions
  \end{itemize}

  \item \textbf{Method C --- OpenIE-style}
  \begin{itemize}
    \item Best recall (0.806) but introduces structural noise
    \item Errors from overly permissive relation scopes
    \item Sensitive to ambiguous or generic relation phrases
  \end{itemize}

  \item \textbf{Method D --- Hybrid}
  \begin{itemize}
    \item Strong precision (0.889), low FP rate
    \item Remaining errors mainly due to:
    \begin{itemize}
      \item long-distance dependencies beyond threshold
      \item missed implicit relations without explicit cue
    \end{itemize}
  \end{itemize}
\end{itemize}

\end{frame}

% ─── DISCUSSION ──────────────────────────────────────────────────────────────

\begin{frame}{Discussion}
\small

\begin{itemize}
  \item \textbf{Key finding: trade-off between precision and recall}
  \begin{itemize}
    \item Rule-based constraints significantly improve precision
    \item However, stricter filtering reduces recall
  \end{itemize}

  \item \textbf{Best overall method: Hybrid (D)}
  \begin{itemize}
    \item Achieves best F1 (0.762) across all methods
    \item Balances explicit linguistic rules with controlled fallback expansion
    \item Strong reduction in false positives compared to C
  \end{itemize}

  \item \textbf{Main bottleneck}
  \begin{itemize}
    \item Recall limited by proximity thresholds and sentence-level extraction
    \item Cross-sentence and long-range dependencies remain unhandled
  \end{itemize}

  \item \textbf{Interpretation}
  \begin{itemize}
    \item A: recall-heavy baseline
    \item B: precision-heavy but brittle system
    \item C: recall-maximising noisy extractor
    \item D: controlled balance between both extremes
  \end{itemize}

\end{itemize}
